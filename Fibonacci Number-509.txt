Fibonacci Number-509
class Solution {
    public int fib(int n) {
        if(n==0 ||n==1)
        {
            return n;
        }
        int var1=0;
        int var2=1;
        int current=0;
        for(int i=2;i<=n;i++)
        {
            current=var1+var2;
            var1=var2;
            var2=current;
        }
        return var2;
    }
}
